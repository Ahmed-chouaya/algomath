   ```python
   import re
   from subprocess import TimeoutExpired
   
   def categorize_error(error: Exception, stderr: str = "") -> ExecutionError:
       """Categorize exception into ExecutionError type."""
       if isinstance(error, TimeoutExpired):
           return ExecutionError.TIMEOUT_ERROR
       if isinstance(error, MemoryError):
           return ExecutionError.MEMORY_ERROR
       if isinstance(error, SyntaxError):
           return ExecutionError.SYNTAX_ERROR
       
       # Check stderr for syntax errors
       if "SyntaxError" in stderr:
           return ExecutionError.SYNTAX_ERROR
       if "MemoryError" in stderr or "out of memory" in stderr.lower():
           return ExecutionError.MEMORY_ERROR
       
       return ExecutionError.RUNTIME_ERROR
   ```

4. **ErrorTranslator class** per D-18, D-20:
   ```python
   class ErrorTranslator:
       """Translate technical errors to mathematician-friendly language."""
       
       TRANSLATIONS = {
           ExecutionError.SYNTAX_ERROR: {
               'message': 'Generated code has a syntax issue',
               'hint': 'This is likely a translation error. Try regenerating with clearer pseudocode.'
           },
           ExecutionError.TIMEOUT_ERROR: {
               'message': 'Algorithm took too long to complete',
               'hint': 'Check for infinite loops or consider optimizing the algorithm.'
           },
           ExecutionError.MEMORY_ERROR: {
               'message': 'Algorithm used too much memory',
               'hint': 'Consider using more memory-efficient data structures or algorithms.'
           },
           ExecutionError.RUNTIME_ERROR: {
               'message': 'Algorithm encountered an error while running',
               'hint': 'Review the technical details below for debugging information.'
           }
       }
       
       @classmethod
       def translate(cls, error: ExecutionError, technical: str = "") -> ErrorDetails:
           translation = cls.TRANSLATIONS.get(error, cls.TRANSLATIONS[ExecutionError.RUNTIME_ERROR])
           return ErrorDetails(
               category=error,
               user_message=translation['message'],
               hint=translation['hint'],
               technical_details=technical if technical else None
           )
   ```

5. **extract_line_number() helper** for debugging per D-19:

   ```python
   def extract_line_number(traceback: str) -> Optional[int]:
       """Extract line number from Python traceback."""
       match = re.search(r'File "[^"]+", line (\d+)', traceback)
       if match:
           return int(match.group(1))
       return None
   ```

   ```python
   MAX_INLINE_LINES = 50
   TRUNCATION_MESSAGE = "\n... ({} more lines - see full log for details)\n"
   
   def truncate_output(text: str, max_lines: int = MAX_INLINE_LINES) -> str:
       """Truncate output to max_lines with summary."""
       lines = text.split('\n')
       if len(lines) <= max_lines:
           return text
       
       truncated = '\n'.join(lines[:max_lines])
       remaining = len(lines) - max_lines
       return truncated + TRUNCATION_MESSAGE.format(remaining)
   ```

2. **ExecutionFormatter class** per D-14, D-16:
   ```python
   from dataclasses import dataclass
   from typing import Optional
   from datetime import datetime
   
   @dataclass
   class FormattedResult:
       display_text: str
       status_emoji: str
       status_text: str
       execution_time: str
       output_truncated: bool
       full_log_path: Optional[str]
   
   class ExecutionFormatter:
       """Format execution results for display."""
       
       def __init__(self, algorithm_name: Optional[str] = None):
           self.algorithm_name = algorithm_name
           self.log_dir = Path(".algomath/algorithms") / algorithm_name if algorithm_name else None
       
       def format_results(self, result: ExecutionResult) -> FormattedResult:
           """Format execution result for display per D-14, D-15, D-16."""
           # Status indicators
           if result.status == 'success':
               emoji = '✓'
               status = 'Success'
           elif result.status == 'timeout':
               emoji = '⏱'
               status = 'Timed Out'
           else:
               emoji = '✗'
               status = 'Failed'
           
           # Build display text
           lines = []
           lines.append(f"{emoji} Execution: {status}")
           lines.append(f"Time: {result.runtime_seconds:.3f}s")
           
           if result.stdout:
               lines.append("\n--- Output ---")
               output = truncate_output(result.stdout, MAX_INLINE_LINES)
               lines.append(output)
           
           if result.stderr:
               lines.append("\n--- Errors ---")
               errors = truncate_output(result.stderr, 20)  # Fewer error lines
               lines.append(errors)
           
           # Full log path per D-14
           log_path = str(self.log_dir / "execution.log") if self.log_dir else None
           if log_path and len(result.stdout.split('\n')) > MAX_INLINE_LINES:
               lines.append(f"\nFull log: {log_path}")
           
           return FormattedResult(
               display_text='\n'.join(lines),
               status_emoji=emoji,
               status_text=status,
               execution_time=f"{result.runtime_seconds:.3f}s",
               output_truncated=len(result.stdout.split('\n')) > MAX_INLINE_LINES if result.stdout else False,
               full_log_path=log_path
           )
   ```

   ```python
   def show_progress(phase: str, current: int, total: int) -> str:
       """Generate progress bar string per Phase 1 D-09."""
       if total <= 0:
           return f"{phase}: ░░░░░░░░░░ 0%"
       
       filled = int(10 * current / total)
       filled = max(0, min(filled, 10))
       bar = '█' * filled + '░' * (10 - filled)
       pct = int(100 * current / total)
       return f"{phase}: {bar} {pct}%"
   ```
