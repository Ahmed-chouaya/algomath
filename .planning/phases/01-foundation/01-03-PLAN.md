```python
import pytest
from .algomath.context import ContextManager
from .algomath.state import WorkflowState

class TestContextManager:
    def test_start_session(self, context_manager):
        cm = context_manager
        cm.start_session()
        state = cm.get_current()
        assert state.current_state == WorkflowState.IDLE
        assert state.current_algorithm is None

    def test_save_text_transitions_state(self, context_manager):
        cm = context_manager
        cm.start_session()
        cm.save_text("test algorithm", name="test_algo")
        state = cm.get_current()
        assert state.current_state == WorkflowState.TEXT_EXTRACTED
        assert state.current_algorithm == "test_algo"

    def test_state_transitions(self, context_manager):
        cm = context_manager
        cm.start_session()
        cm.save_text("text")
        assert cm.get_current().current_state == WorkflowState.TEXT_EXTRACTED
        
        cm.save_steps([{"step": 1}])
        assert cm.get_current().current_state == WorkflowState.STEPS_STRUCTURED
        
        cm.save_code("def algo(): pass")
        assert cm.get_current().current_state == WorkflowState.CODE_GENERATED

    def test_session_persistence(self, context_manager, temp_algopath):
        cm = context_manager
        cm.start_session()
        cm.save_text("persistent text")
        
        # Simulate new session
        cm2 = ContextManager(base_path=temp_algopath)
        cm2.start_session()
        state = cm2.get_current()
        assert state.data.get("text") == "persistent text"

    def test_get_progress(self, context_manager):
        cm = context_manager
        cm.start_session()
        progress = cm.get_progress()
        assert "state" in progress
        assert "percent" in progress
```

```python
import pytest
from src.intent import detect_intent, IntentType, route_to_workflow

class TestIntentDetection:
    def test_extract_keywords(self):
        inputs = [
            "extract the algorithm",
            "parse this text",
            "get steps from this",
            "analyze this algorithm"
        ]
        for inp in inputs:
            intent, conf = detect_intent(inp)
            assert intent == IntentType.EXTRACT
            assert conf > 0.5

    def test_generate_keywords(self):
        inputs = [
            "generate code",
            "create python code",
            "write the implementation"
        ]
        for inp in inputs:
            intent, conf = detect_intent(inp)
            assert intent == IntentType.GENERATE
            assert conf > 0.5

    def test_run_keywords(self):
        inputs = [
            "run the code",
            "execute this",
            "test the algorithm"
        ]
        for inp in inputs:
            intent, conf = detect_intent(inp)
            assert intent == IntentType.RUN
            assert conf > 0.5

    def test_verify_keywords(self):
        inputs = [
            "verify results",
            "check output",
            "validate this"
        ]
        for inp in inputs:
            intent, conf = detect_intent(inp)
            assert intent == IntentType.VERIFY
            assert conf > 0.5

    def test_list_keywords(self):
        intent, conf = detect_intent("show me my algorithms")
        assert intent == IntentType.LIST
        assert conf > 0.5

    def test_help_keywords(self):
        intent, conf = detect_intent("how do I use this")
        assert intent == IntentType.HELP
        assert conf > 0.5

    def test_unknown_input(self):
        intent, conf = detect_intent("random unrelated text")
        assert intent == IntentType.UNKNOWN
        assert conf < 0.5

    def test_confidence_calculation(self):
        # Single keyword should have lower confidence than multiple
        _, conf1 = detect_intent("extract")

        assert conf2 > conf1

class TestRouteToWorkflow:

        assert path == "src.workflows.generate"

    def test_run_routing(self):
        path = route_to_workflow(IntentType.RUN)
        assert path == "src.workflows.run"
```

```python
import pytest
from src.context import ContextManager
from src.workflows import run_extraction
from src.intent import detect_intent, IntentType

class TestEndToEndWorkflow:
    """Test complete user workflows from start to finish."""

    @pytest.fixture
    def fresh_session(self, temp_algopath):
        cm = ContextManager(base_path=temp_algopath)
        cm.start_session()
        return cm

    def test_full_extraction_workflow(self, fresh_session):
        """User extracts algorithm, system persists it."""
        cm = fresh_session
        
        # Step 1: Extract intent
        intent, conf = detect_intent("extract this algorithm")
        assert intent == IntentType.EXTRACT
        assert conf > 0.5
        
        # Step 2: Run extraction
        result = run_extraction(cm, text="Algorithm: Sort an array")
        assert result["status"] in ["success", "stub"]
        
        # Step 3: Verify persistence
        assert cm.get_current().data["text"] == "Algorithm: Sort an array"
        
        # Step 4: Check state
        from src.state import WorkflowState
        assert cm.get_current().current_state == WorkflowState.TEXT_EXTRACTED

    def test_session_resumption(self, fresh_session, temp_algopath):
        """Simulate interruption and resumption."""
        cm = fresh_session
        
        # User starts work
        cm.save_text("My algorithm", name="my_algo")
        cm.save_steps([{"step": "Initialize"}])
        
        # Simulate crash - new ContextManager
        cm2 = ContextManager(base_path=temp_algopath)
        cm2.start_session()
        
        # Verify state recovered
        state = cm2.get_current()
        assert state.data["text"] == "My algorithm"
        assert len(state.data["steps"]) == 1

    def test_named_vs_unnamed_algorithms(self, fresh_session):
        """Test both named (persisted) and unnamed (session-only) algorithms."""
        cm = fresh_session
        
        # Unnamed - session only
        cm.save_text("Unnamed algo")
        assert cm.get_current().current_algorithm is None
        
        # Named - persisted
        cm.create_algorithm("named_algo")
        cm.save_text("Named algo text")
        assert cm.get_current().current_algorithm == "named_algo"

    def test_progress_indicators(self, fresh_session):
        """Verify progress display at different states."""
        cm = fresh_session
        
        # Initial state
        progress = cm.get_progress()
        assert progress["percent"] == 0
        
        # After extraction
        cm.save_text("text")
        progress = cm.get_progress()
        assert progress["percent"] > 0
        
        # After steps
        cm.save_steps([{"step": 1}])
        progress = cm.get_progress()
        assert progress["percent"] > 25
```
